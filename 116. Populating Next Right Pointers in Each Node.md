class Solution {
    public Node connect(Node root) {
        if (root == null || root.left == null) return root;
        Node node = root;
        while (node != null) {
            Node curr = node;
            if (curr.left != null) {
                while (curr != null) {
                    curr.left.next = curr.right;
                    if(curr.next != null) curr.right.next = curr.next.left;
                    curr = curr.next;
                }
            }
            node = node.left;
        }
        return root;
	}
}
