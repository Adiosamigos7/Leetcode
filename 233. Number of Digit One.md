class Solution {
    public int countDigitOne(int n) {
        if (n <= 0) return 0;
        long ones = 0;
        for (long i = 1, q = n; i <= n; i *= 10, q /= 10) {
            long prefix = n / (i * 10), curr = q % 10, suffix = n % i;
            ones += prefix * i;
            ones += 1 < curr ? i : (1 == curr ? suffix + 1 : 0);
        }
        return (int) ones;
    }
}
